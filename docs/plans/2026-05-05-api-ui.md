## Goal

Expose the dispatcher to the outside world. After this plan:
- Existing `tika-server` clients can repoint to RedTusk with zero code changes.
- Users can POST files to `/v1/jobs`, poll status, and fetch artifacts.
- A single-page UI shows recent submissions.
- `redtusk serve` starts the full stack.
- `redtusk version` and `redtusk selftest` work.

## Tasks

### Task 1 — Translation layer (`translation.py`)

Pure functions; no I/O; no asyncio. Takes an `ExtractResult` and produces
the wire-format body for each tika-compat endpoint.

```python
def to_tika_text(result: ExtractResult) -> str:
    """PUT /tika → text/plain. Root entry text or empty string."""
    # result.extraction.entries[0].text (root entry, depth=0)

def to_meta_json(result: ExtractResult) -> dict[str, Any]:
    """PUT /meta → application/json. Root entry metadata dict only.

    Tika /meta returns a single flat dict of metadata keys from the root entry.
    """
    # result.extraction.entries[0].metadata  (already a dict)

def to_rmeta_json(result: ExtractResult) -> list[dict[str, Any]]:
    """PUT /rmeta → application/json. All entries in tika-server format.

    Tika rmeta returns a list where each item is a flat dict:
      - All metadata keys from entry.metadata (string values)
      - "X-TIKA:content" → entry.text (if non-empty)
      - "Content-Type" → entry.content_type
      - "X-TIKA:embedded_resource_path" → entry.path (for depth > 0)
    """

def to_detect_text(result: ExtractResult) -> str:
    """PUT /detect/stream → text/plain. Root content type."""
    # result.extraction.root_content_type

def to_language_text(result: ExtractResult) -> str:
    """PUT /language/stream → text/plain. Root language or empty."""
    # result.extraction.root_language or ""

def to_unpack_tar(result: ExtractResult, *, include_root: bool) -> bytes:
    """PUT /unpack and /unpack/all → application/x-tar.

    Build an in-memory tar archive. Each entry with non-empty bytes
    is added as a file named by its path (stripping leading /). Root
    entry is included only when include_root=True.

    Entry bytes come from entry.sha256 lookup — we don't have the raw
    bytes in ExtractResult. Return empty tar for now (placeholder).
    See note below.
    """
```

**Note on /unpack:** Tika's `/unpack` returns actual file bytes. Our worker writes
per-entry text files to `out/`, not raw bytes. The unpack endpoint is therefore
**a best-effort stub** in v1: return a tar containing per-entry text if available,
or an empty tar. Add a `Warning: unpack-bytes-unavailable` response header.
Full binary unpack (writing raw embedded bytes from the worker) is a v2 feature.

**to_rmeta_json format detail:** Tika rmeta entries look like:
```json
[
  {
    "Content-Type": "application/pdf",
    "Author": "Jane Smith",
    "X-TIKA:content": "extracted text here",
    "X-TIKA:embedded_resource_path": "/embedded-0.png"
  }
]
```
Produce this shape by:
1. Start with a copy of `entry.metadata` (already a `dict[str, str]`).
2. Set `"Content-Type"` to `entry.content_type`.
3. If `entry.text`: set `"X-TIKA:content"` to `entry.text`.
4. If `entry.depth > 0`: set `"X-TIKA:embedded_resource_path"` to `entry.path`.

**Tests:** `tests/http/test_tika_compat.py` tests the translation functions
indirectly via the API. No dedicated translation unit test file needed — the
API tests provide sufficient coverage.

```python
def create_app(
```

#### Tika-compat endpoints

All share a single `_run_sync(body, filename, limits, dispatcher)` helper that calls
`dispatcher.submit_sync(body, filename, limits)` and catches:
- `PoolExhaustedError` → 503 with `Retry-After: 5` header
- `WorkerError` → 502
- `SchemaValidationError` → 502

**Endpoints:**

```python

@app.put("/rmeta/html")

@app.put("/unpack/all")
async def put_unpack_all(request: Request) -> Response:
    ...include_root=True...

@app.get("/version")
async def get_version() -> PlainTextResponse:
    return PlainTextResponse(f"Apache Tika 3.3.0 (RedTusk {__version__})")

@app.get("/mime-types")
async def get_mime_types() -> JSONResponse:
    # Return a static dict of common MIME types (just enough for client compat)
    # Actual Tika /mime-types returns a large JSON; we return a representative subset.
    return JSONResponse(MIME_TYPES_STUB)

    return JSONResponse(PARSERS_STUB)
```

`MIME_TYPES_STUB` and `PARSERS_STUB` are module-level dicts — small static stubs
that satisfy client compat checks without running Java. Include at minimum:
- `MIME_TYPES_STUB`: `{"application/pdf": {"alias": [], "supertype": "application/octet-stream"}, ...}` — 10-20 common types
- `PARSERS_STUB`: `{"name": "org.apache.tika.parser.AutoDetectParser", "children": []}`

#### Readiness, metrics, UI

```python
@app.get("/v1/readyz")
async def readyz(request: Request) -> JSONResponse:
    if request.app.state.dispatcher.is_healthy():
        return JSONResponse({"status": "ok"})
    return JSONResponse({"status": "degraded"}, status_code=503)

@app.get("/metrics")
async def metrics() -> Response:
    body, content_type = render_for_endpoint()
    return Response(content=body, media_type=content_type)

@app.get("/")
async def ui() -> HTMLResponse:
    return HTMLResponse(content=_load_ui())
```

`_load_ui()` reads `static/index.html` relative to the `api.py` file:
```python
_STATIC = Path(__file__).parent / "static"
def _load_ui() -> str:
    return (_STATIC / "index.html").read_text()
```

#### Lifespan (startup/shutdown)

```python
from contextlib import asynccontextmanager

@asynccontextmanager
async def lifespan(app: FastAPI):
    await app.state.dispatcher.start()
    yield
    await app.state.dispatcher.stop()
```

Pass `lifespan=lifespan` to `FastAPI(...)`.

### Task 3 — Static UI (`static/index.html`)

A minimal single-page app. No build step, no npm — pure HTML + vanilla JS.

Design requirements:
- Title: "RedTusk" with red/tusk accent color.
- Page body: a table of recent jobs (job ID, filename, state, submitted_at, duration).
- Auto-refresh every 5 seconds via `setInterval(fetchJobs, 5000)`.
- `fetchJobs()` calls `GET /v1/jobs?limit=50` and re-renders the table.
- Clicking a job ID row expands an inline JSON view of the job result (rmeta).
- A file upload form at the top: `<input type="file">` + submit button → `PUT /rmeta` → shows result inline.
- No external CDN dependencies (works offline).
- Responsive layout, dark-friendly, < 250 lines of HTML/CSS/JS combined.

Color palette: dark background `#1a1a1a`, accent red `#c0392b`, text `#e0e0e0`.

### Task 4 — CLI (`cli.py`)

```python
"""RedTusk command-line interface."""
import asyncio
import sys
import click

@click.group()
def cli() -> None:
    """RedTusk — sandboxed Apache Tika service."""

@cli.command()
@click.option("--host", default="0.0.0.0", show_default=True)
@click.option("--port", default=8000, show_default=True, type=int)
@click.option("--workers", default=1, show_default=True, type=int)
def serve(host: str, port: int, workers: int) -> None:
    """Start the RedTusk API server with embedded dispatcher."""
    import uvicorn
    from redtusk.limits import Limits
    from redtusk.jobs.memory import MemoryJobStore
    from redtusk.pool import Pool
    from redtusk.worker_runtime import DockerWorkerRuntime
    from redtusk.runtime.docker_runtime import DockerRuntime
    from redtusk.dispatcher import Dispatcher
    from redtusk.api import create_app
    from redtusk.observability.logging import configure_logging

    configure_logging()
    limits = Limits.from_env()
    store = MemoryJobStore()
    # DockerRuntime.detect() is async; run it synchronously at startup
    docker_rt = asyncio.run(DockerRuntime.detect())
    image = os.environ.get("REDTUSK_WORKER_IMAGE", "redtusk:latest")
    worker_rt = DockerWorkerRuntime(docker=docker_rt, limits=limits, image=image)
    pool = Pool(limits=limits, worker_runtime=worker_rt, store=store, profile=limits.profile)
    dispatcher = Dispatcher(pool=pool, store=store, worker_runtime=worker_rt, limits=limits)
    app = create_app(dispatcher=dispatcher, store=store, limits=limits)
    uvicorn.run(app, host=host, port=port, workers=workers)

@cli.command()
def version() -> None:
    """Print version and exit."""
    from redtusk._version import __version__
    click.echo(f"RedTusk {__version__} (Tika 3.3.0)")

@cli.command()
def selftest() -> None:
    """Run a quick self-test (imports and limits validation)."""
    from redtusk.limits import Limits
    from redtusk.schema import validate_rmeta
    try:
        limits = Limits.from_env()
        click.echo(f"Limits OK: pool_size={limits.pool_size}")
        click.echo("Self-test passed.")
    except Exception as e:
        click.echo(f"Self-test FAILED: {e}", err=True)
        sys.exit(1)
```

Register `cli` as the entry point in `pyproject.toml`:
```toml
[project.scripts]
redtusk = "redtusk.cli:cli"
```

Add `click` and `uvicorn` to `[project.dependencies]` if not already present.
Add `fastapi[standard]` (includes starlette, uvicorn).
